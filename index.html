const SHEET_USERS = 'Users'; const SHEET_TX = 'Transactions'; function doGet(e) { return ContentService.createTextOutput("極簡記帳 API 伺服器運作中...") .setMimeType(ContentService.MimeType.TEXT); } function doPost(e) { const output = (data) => ContentService.createTextOutput(JSON.stringify(data)).setMimeType(ContentService.MimeType.JSON); try { const params = JSON.parse(e.postData.contents); const action = params.action; if (action === 'verifyLogin') { return output(verifyLogin(params.user, params.pass)); } else if (action === 'getTransactions') { // 支援 limit 與 offset 的分頁請求 return output(getTransactions(params.user, params.limit, params.offset)); } else if (action === 'saveTransaction') { return output(saveTransaction(params.user, params.tx)); } else if (action === 'deleteTransaction') { return output(deleteTransaction(params.user, params.id)); } else if (action === 'getSummary') { // 用於主畫面快速顯示置中的總結餘 return output(getSummary(params.user)); } else { return output({success: false, message: '無效指令'}); } } catch (err) { return output({success: false, message: 'API 錯誤: ' + err.toString()}); } } // 驗證登入 function verifyLogin(user, pass) { const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(SHEET_USERS); const data = sheet.getDataRange().getDisplayValues(); for (let i = 1; i < data.length; i++) { if (data[i][0] === user && data[i][1] === pass) return { success: true }; } return { success: false, message: '帳號或密碼錯誤' }; } // 取得總結餘 (效能優化：只計算數值，不傳輸明細) function getSummary(user) { const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(SHEET_TX); const data = sheet.getDataRange().getDisplayValues(); let balance = 0; for (let i = 1; i < data.length; i++) { if (data[i][1] === user) { const type = data[i][2]; const amount = parseInt(data[i][4], 10) || 0; balance += (type === 'income' ? amount : -amount); } } return { success: true, balance: balance }; } // 取得帳目 (實作伺服器端分頁與排序) function getTransactions(user, limit = 20, offset = 0) { const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(SHEET_TX); const data = sheet.getDataRange().getDisplayValues(); let userTx = []; for (let i = 1; i < data.length; i++) { if (data[i][1] === user) { userTx.push({ id: data[i][0], type: data[i][2], name: data[i][3], amount: parseInt(data[i][4], 10) || 0, date: data[i][5], time: data[i][6], note: data[i][7] }); } } // 依日期時間倒序排列 (最新的在前) userTx.sort((a, b) => { const timeA = new Date(a.date.replace(/-/g, '/') + ' ' + a.time).getTime(); const timeB = new Date(b.date.replace(/-/g, '/') + ' ' + b.time).getTime(); return timeB - timeA; }); const totalCount = userTx.length; const pagedData = userTx.slice(offset, offset + limit); return { success: true, data: pagedData, hasMore: (offset + limit) < totalCount, totalCount: totalCount }; } // 儲存帳目 (新增或修改) function saveTransaction(user, tx) { const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(SHEET_TX); const timestamp = new Date().toLocaleString('zh-TW', { timeZone: 'Asia/Taipei' }); if (tx.id) { const data = sheet.getDataRange().getDisplayValues(); for (let i = 1; i < data.length; i++) { if (data[i][0] === tx.id && data[i][1] === user) { sheet.getRange(i + 1, 3, 1, 7).setValues([[ tx.type, tx.name, tx.amount, tx.date, tx.time, tx.note || '', timestamp ]]); return { success: true }; } } } const newId = new Date().getTime().toString(); sheet.appendRow([newId, user, tx.type, tx.name, tx.amount, tx.date, tx.time, tx.note || '', timestamp]); return { success: true }; } // 刪除帳目 function deleteTransaction(user, id) { const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(SHEET_TX); const data = sheet.getDataRange().getDisplayValues(); for (let i = 1; i < data.length; i++) { if (data[i][0] === id && data[i][1] === user) { sheet.deleteRow(i + 1); return { success: true }; } } return { success: false, message: '找不到資料' }; }
